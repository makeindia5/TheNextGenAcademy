// â”€â”€ Intersection Observer hook â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
function useFadeUp() {
  const refs = useRef([]);
  useEffect(() => {
    const obs = new IntersectionObserver(
      (entries) => entries.forEach(e => { if (e.isIntersecting) e.target.classList.add("vis"); }),
      { threshold: 0.1 }
    );
    refs.current.forEach(el => { if (el) obs.observe(el); });
    return () => obs.disconnect();
  }, []);
  const addRef = (el) => { if (el && !refs.current.includes(el)) refs.current.push(el); };
  return addRef;
}

    if (course) return <CourseDetailPage courseId={page} />;
    return <HomePage />;
  };

  return (
    <RouterCtx.Provider value={{ page, navigate }}>
      <style>{GLOBAL_CSS}</style>
      <Navbar />
      <main style={{ paddingTop: 70 }}>
        {renderPage()}
      </main>
      <Footer />
      <WhatsAppBtn />
    </RouterCtx.Provider>
